  var carts = [];
 
  var cartsService = {
    purchase: function () {
      var _carts = [];
      for (var isbn in carts) {
        var item = {
          'isbn': isbn,
          'count': carts[isbn]
        };
        _carts.push(item);
      }
      return _carts;
    },
    get: function () {
      console.log('CartsService', carts);

    },

      if (!carts[isbn]) {
        carts[isbn] = 1;
      } else {
        carts[isbn] = carts[isbn] + 1;
      }
    },
    remove: function (isbn) {
      if (!carts[isbn] || carts[isbn] === 0) {
        return ;
      }
 
      carts[isbn] = carts[isbn] - 1;
      if (carts[isbn] === 0) {

      }
    },
    clear: function () {
      carts = [];
    }
  };
